    set_real_ip_from 2a06:98c0::/29;
    real_ip_header CF-Connecting-IP;

    # WordPress configuration
    location / {
        try_files $uri $uri/ /index.php?$args;
    }

    # Configuración de PHP
    location ~ \.php$ {
        include snippets/fastcgi-php.conf;
        fastcgi_pass unix:/var/run/php/php8.1-fpm.sock;
        fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
        fastcgi_param HTTPS on;
        include fastcgi_params;
        # Headers específicos para Cloudflare
        fastcgi_param HTTP_CF_CONNECTING_IP $http_cf_connecting_ip;
        fastcgi_param HTTP_CF_IPCOUNTRY $http_cf_ipcountry;
        fastcgi_param HTTP_CF_RAY $http_cf_ray;
        fastcgi_param HTTP_CF_VISITOR $http_cf_visitor;
    }

    # Seguridad
    location ~ /\.ht {
        deny all;
    }

    location = /favicon.ico {
        log_not_found off;
        access_log off;
    }

    location = /robots.txt {
        allow all;
        log_not_found off;
        access_log off;
    }

    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg)$ {
        expires max;
        log_not_found off;
        access_log off;
        add_header Access-Control-Allow-Origin "*";
    }

    # Headers básicos de seguridad (deja que Cloudflare maneje el resto)
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
}
